Bayra - is a Creative Portfolio Template suitabe for Resume or Portfolio websites designed for creative designers, developers, freelancers or pretty much any profession. This Template is 100% responsive, well organized and very easy to customize.

It's available in Light, Dark and Blue Versions. Bayra is better way to present your modern Resume/Portfolio.

Then open the assets/css/main/homepages.css file using any text editor then look for the following code and replace blue highlighted text with your new image's path, name and extension like this url('../../images/home.jpg'). 

#header {

To edit the progress percentage replace blue highlighted text with the new value. 

// ========================================================================= //
//      [ Skills Coounter ]
// ========================================================================= //


    function skillsProgress() {
        var lang = {
            "html": "100%",
            "css": "90%",
            "javascript": "70%"
        };

        var multiply = 2;

        $.each(lang, function(language, pourcent) {

            var delay = 600;

            setTimeout(function() {
                $('#' + language + '-pourcent').html(pourcent);
                $('#progress-' + language).animate({
                    'width': pourcent
                }, 600);
            }, delay * multiply);

            multiply++;

        });
    }
